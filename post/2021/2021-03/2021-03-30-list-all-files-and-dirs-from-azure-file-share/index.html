/*
retrieve all files and dics from  Azure File share
*/

async function retrieveFileShare() {
            let index = 0;
            reportStatus("Retrieving file shares list...");
            try {
                console.log(`List shares`);
                let shareIter = serviceClientWithSAS.listShares();
                let i = 1;
                for await (const share of shareIter) {
                    console.log(`Share ${i}: ${share.name}`);
                    i++;
                    index += 1;

                    let fileShareFromStorage = {
                        value: share.name,
                        label: share.name
                    }
                    shareList.push(fileShareFromStorage);
                }

                console.log('shareList: ', shareList);

                if (index > 0) {
                    reportStatus("Done.");
                } else {
                    reportStatus("The storage account does not contain any file shares.");
                }
            } catch (error) {
                reportStatus(error.message);
            }

            if (!unmounted) {
                console.log("shareList: ", shareList);
                setFileShareItems(
                    shareList
                );
                setLoading(false);
            }
        }
        retrieveFileShare();

        async function retrieveDirsFromFileShare() {
            const directoryName = '.';
            const shareClient = serviceClientWithSAS.getShareClient(selectedFileShare);
            const dirClient = shareClient.getDirectoryClient(directoryName);
            let index = 0;
            reportStatus("Retrieving file list...");
            try {
                let entityIter = dirClient.listFilesAndDirectories();
                let i = 1;
                for await (const entity of entityIter) {
                    index += 1;
                    if (entity.kind === "directory") {
                        console.log(`${i++} - directory\t: ${entity.name} path\t: ${JSON.stringify(entity)}`);
                        let dirFromStorage = {
                            value: entity.name,
                            label: entity.name
                        }
                        dirList.push(dirFromStorage);
                      }
                }
                console.log('dirList: ', dirList);

                if (index > 0) {
                    reportStatus("Done.");
                } else {
                    reportStatus("The file share does not contain any files.");
                }
            } catch (error) {
                console.log(error.message);
            }
            if (!unmounted) {
                console.log("shareList: ", shareList);
                setItems(
                    fileList
                );
                setDirItems(
                    dirList
                );
                setLoading(false);
            }
        }

        retrieveDirsFromFileShare()

        async function retrieveFilesFromDir(directoryName) {
            const shareClient = serviceClientWithSAS.getShareClient(selectedFileShare);
            const dirClient = shareClient.getDirectoryClient(directoryName);
            let index = 0;
            reportStatus("Retrieving file list...");
            try {
                let entityIter = dirClient.listFilesAndDirectories();
                let i = 1;
                for await (const entity of entityIter) {
                    index += 1;
                    if (entity.kind === "file") {
                        console.log(`${i++} - file\t: ${entity.name} - path\t: ${JSON.stringify(entity)}`);
                        let fileFromStorage = {
                            value: entity.name,
                            label: entity.name
                        }
                        fileList.push(fileFromStorage);
                      }
                }
                console.log('dirList: ', dirList);

                console.log('fileList: ', fileList);
                if (index > 0) {
                    reportStatus("Done.");
                } else {
                    reportStatus("The file share does not contain any files.");
                }
            } catch (error) {
                console.log(error.message);
            }
            if (!unmounted) {
                console.log("shareList: ", shareList);
                setItems(
                    fileList
                );
                setDirItems(
                    dirList
                );
                setLoading(false);
            }
        }

        retrieveFilesFromDir(selectedDir)

        return () => {
            unmounted = true;
        };
    }, [selectedFileShare, selectedDir]);
